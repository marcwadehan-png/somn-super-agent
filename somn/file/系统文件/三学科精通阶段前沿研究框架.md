```
LTV = Σ(t=1 to n) [ARPU × 留存率(t) × 毛利率(t)] / (1 + d)^t

- ARPU: 每用户平均收入

- 毛利率: 随规模变化的成本曲线
- d: 折现率
```

```python
        self.slots = {
            'intent': None,      # 用户意图

            'current_intent': self.slots['intent'],
```

**A/B测试的统计设计**：

```python
class ABTestDesigner:
    """A/B测试设计器"""
    
    def __init__(self, baseline_rate, mde, alpha=0.05, power=0.8):
        self.baseline = baseline_rate  # 基线转化率
        self.mde = mde                 # 最小可检测效应
        self.alpha = alpha             # 第一类错误率
        self.power = power             # 统计功效
        
    def calculate_sample_size(self):
        """计算样本量"""
        z_alpha = norm.ppf(1 - self.alpha/2)
        z_beta = norm.ppf(self.power)
        
        p1 = self.baseline
        p2 = self.baseline * (1 + self.mde)
        
        n = (2 * (z_alpha + z_beta)**2 * 
             (p1*(1-p1) + p2*(1-p2))) / (p2 - p1)**2
        
        return int(n)
    
    def sequential_test(self, observations):
        """序贯概率比检验"""
        likelihood_ratio = self._compute_lr(observations)
        
        if likelihood_ratio > 1/self.alpha:
            return {'decision': 'reject_H0', 'evidence': 'significant'}
        elif likelihood_ratio < self.power:
            return {'decision': 'continue', 'evidence': 'inconclusive'}
        else:
            return {'decision': 'accept_H0', 'evidence': 'not_significant'}
```

```python
class LiquidityTrapAnalyzer:
    """流动性陷阱分析器"""
    
    def __init__(self, data):
        self.data = data
    
    def detect_liquidity_trap(self):
        """识别流动性陷阱"""
        indicators = {
            'real_rate_near_zero': self.data['real_rate'] < 0.5,
            'inflation_low': self.data['inflation'] < 1.0,
            'qe_active': self.data['qe_balance_sheet'] > 0,

            'in_trap': trap_score > 0.6,
            'severity': trap_score,
            'indicators': indicators
```

**算法公平性框架**：

```python
class FairnessAnalyzer:
    """公平性分析器"""
    
    # 公平性定义
    FAIRNESS_CRITERIA = {
        'demographic_parity': lambda pred, sensitive: (
            pred[sensitive==0].mean() == pred[sensitive==1].mean()
        ),
        'equalized_odds': lambda pred, sensitive, label: (
            pred[(sensitive==0)&(label==1)].mean() == 
            pred[(sensitive==1)&(label==1)].mean() and
            pred[(sensitive==0)&(label==0)].mean() == 
            pred[(sensitive==1)&(label==0)].mean()
        ),
        'calibration': lambda pred, label: (
            # 相同预测概率下，不同群体的真实概率相同
            p(y=1|pred=p, s=0) == p(y=1|pred=p, s=1) for all p
        )
    }
    
    def evaluate_fairness(self, predictions, labels, sensitive_attr):
        """评估多种公平性指标"""
        results = {}
        for name, criteria in self.FAIRNESS_CRITERIA.items():
            try:
                results[name] = criteria(predictions, sensitive_attr, labels)
            except:
                results[name] = None
        return results
    
    def fairness_utility_tradeoff(self, predictions, labels, sensitive_attr):
        """公平性与效用权衡"""
        # 不公平度 = Disparate Impact
        di = (predictions[sensitive_attr==1].mean() / 
              predictions[sensitive_attr==0].mean())
        
        # 效用损失 = 准确率下降
        baseline_acc = accuracy_score(labels, predictions)
        
        return {
            'disparate_impact': di,
            'baseline_accuracy': baseline_acc,
            'tradeoff_ratio': (1 - di) / baseline_acc
        }
```

```
SROI = 社会价值产出 / 社会投资投入

         Σ (产出 × 成果权重) / 成果单位价值
SROI = ─────────────────────────────────────────
        Σ (投资 + 运营成本 + 时间成本)

案例：微型金融项目
- 贷款总额：1000万
- 受益人数：5000人
- 就业创造：2000个岗位
- 收入提升：平均30%

SROI = (2000×就业价值 + 5000×收入提升值 + ...) / 1000万
     = 3.5:1

含义：每1元社会投资产生3.5元社会价值
```

```python
class SignalingGameAnalyzer:
    """信号博弈分析器"""
    
    def __init__(self):
        self.game_elements = {
            'sender': {
                'types': ['high_quality', 'low_quality'],  # 类型
                'cost': {'high': 10, 'low': 50},             # 信号成本
                'actions': ['send_signal', 'no_signal']
            },
            'receiver': {
                'actions': ['accept', 'reject'],
                'utility': {
                    'high_quality_accept': 100,
                    'low_quality_accept': -50,
                    'reject': 0
                }
            }
        }
    
    def find_separating_equilibrium(self):
        """分离均衡分析"""
        # 分离均衡条件：
        # 高质量类型：发送信号，接收方接受
        # 低质量类型：不发送信号，接收方拒绝
        
        conditions = {
            'sender_incentive': {
                'high_type': '发送信号的净收益 > 不发送',
                'low_type': '不发送信号的净收益 > 发送'
            },
            'receiver_belief': {
                'with_signal': 'P(high|type) = 1',
                'without_signal': 'P(low|type) = 1'
            }
        }
        
        # 均衡存在条件：信号成本差异足够大
        cost_diff = self.game_elements['sender']['cost']['low'] - \
                   self.game_elements['sender']['cost']['high']
        
        equilibrium_exists = cost_diff > 40
        
        return {
            'exists': equilibrium_exists,
            'conditions': conditions,
            'threshold': 40
        }
    
    def find_pooling_equilibrium(self):
        """混合均衡分析"""
        # 混合均衡：两种类型都发送相同信号
        
        conditions = {
            'both_send': {
                'receiver': '接受（因为无法区分）',
                'high_type': '发送的预期收益 ≥ 不发送',
                'low_type': '发送的预期收益 ≥ 不发送'
            },
            'both_no_signal': {
                'receiver': '拒绝（因为质量不确定）',
                'high_type': '不发送的预期收益 ≥ 发送',
                'low_type': '不发送的预期收益 ≥ 发送'
            }
        }
        
        return conditions
    
    def evaluate_signaling_efficiency(self, signal_cost, quality_differential):
        """评估信号效率"""
        # 社会最优信号量
        # 信号过少：逆向选择导致市场萎缩
        # 信号过多：资源浪费
        
        efficiency_ratio = quality_differential / signal_cost
        
        return {
            'efficient_signaling': efficiency_ratio > 1,
            'over_signaling': signal_cost > quality_differential,
            'social_waste': max(0, signal_cost - quality_differential)
        }
```

#### 15.3 声誉机制与信任建立

**重复博弈中的声誉效应**：

```
声誉资本的价值：

声誉价值 = Σ 未来合作收益 × 声誉溢价系数

声誉建立机制：

一次性博弈（无声誉价值）：
├─ 欺骗的诱惑：P(欺骗) × 短期收益
├─ 合作收益：0
└─ 结果：普遍欺骗

无限期重复博弈（有声誉价值）：
├─ 贴现因子δ → 合作收益的现值
├─ 触发策略：一旦欺骗，永远不合作
├─ 合作条件：δ ≥ 欺骗收益 / 合作收益
└─ 结果：声誉投资→合作均衡

有限期重复博弈的声誉诅咒：
```

```python
        pseudo_outcomes = []
        for fold in folds:

            train_idx = np.setdiff1d(range(len(Y)), fold)
            
            # 预测Y
            model_y = self._train_ml_model(X[train_idx], Y[train_idx])
```

```python
        }
        
        # 量化的潜在影响
        quantitative_impact = {
            'bank_deposits_shift': self._estimate_deposit_shift(

        }

            'quantitative_impact': quantitative_impact,

    def design_privacy_framework(self):
```

```python
class ResearchDesignGoldStandard:
    """学术研究设计黄金标准"""
    
    DESIGN_CHECKLIST = {
        'research_question': {
            'importance': '是否重要（理论/实践）',
            'specificity': '是否具体可测',
            'novelty': '是否有边际贡献',
            'feasibility': '是否可操作检验'
        },
        'theory_development': {
            'mechanism': '因果机制是否清晰',
            'boundary': '理论边界条件是否明确',
            'testable_hypotheses': '是否推导出可检验假设',
            'alternative_explanations': '是否考虑了替代解释'
        },
        'identification_strategy': {
            'causal_identification': '因果识别策略是否可信',
            'identification_assumptions': '识别假设是否明确且可辩护',
            'internal_validity': '内部效度威胁是否处理',
            'external_validity': '外部效度范围是否界定'
        },
        'data_quality': {
            'measurement': '测量是否有效可靠',
            'sample_selection': '样本选择是否存在偏误',
            'data_cleaning': '数据清洗是否透明',
            'reproducibility': '数据是否可复现'
        },
        'estimation_strategy': {
            'model_specification': '模型设定是否合理',
            'robustness': '是否有稳健性检验',
            'heterogeneity': '是否探索异质性效应',
            'sensitivity': '是否进行敏感性分析'
        },
        'reporting_standards': {
            'effect_sizes': '是否报告效应量',
            'confidence_intervals': '是否报告置信区间',
            'limitations': '是否坦诚讨论局限性',
            'open_science': '代码/数据是否公开'
        }
    }
    
    def evaluate_research_quality(self, paper_components):
        """评估研究质量"""
        scores = {}
        for dimension, checklist in self.DESIGN_CHECKLIST.items():
            scores[dimension] = self._rate_checklist(
                paper_components.get(dimension, {}),
                checklist
            )
        
        # 整体质量评估
        overall_score = np.mean(list(scores.values()))
        
        # 薄弱环节识别
        weak_areas = [
            dim for dim, score in scores.items() 
            if score < 0.6
        ]
        
        return {
            'scores': scores,
            'overall': overall_score,
            'weak_areas': weak_areas,
            'recommendations': self._suggest_improvements(weak_areas)
        }
```
