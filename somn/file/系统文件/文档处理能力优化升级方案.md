### 4.1 自然语言处理接口

```python
# src/documents/nlp_document_service.py
from typing import Dict, Any, List, Optional
from pathlib import Path
from enum import Enum

class DocumentIntent(Enum):
    """文档处理意图"""
    GENERATE = "生成"
    READ = "读取"
    MODIFY = "修改"
    CONVERT = "转换"
    MERGE = "合并"
    ANALYZE = "分析"

class NLPDocumentService:
    """自然语言文档处理服务"""
    
    def __init__(self):
        self.intent_classifier = self._load_intent_classifier()
        self.ppt_service = self._init_ppt_service()
        self.docx_generator = self._init_docx_generator()
        self.excel_generator = self._init_excel_generator()
    
    def process_command(self, command: str) -> Dict[str, Any]:
        """处理自然语言文档指令"""
        # 1. 意图识别
        intent = self._classify_intent(command)
        
        # 2. 参数提取
        params = self._extract_params(command, intent)
        
        # 3. 执行处理
        result = self._execute_intent(intent, params)
        
        return result
    
    def _classify_intent(self, command: str) -> DocumentIntent:
        """识别处理意图"""
        # 意图关键词映射
        intent_keywords = {
            DocumentIntent.GENERATE: ["生成", "创建", "制作", "做"],
            DocumentIntent.READ: ["读取", "打开", "查看", "分析"],
            DocumentIntent.MODIFY: ["修改", "编辑", "添加", "更新"],
            DocumentIntent.CONVERT: ["转换", "转成", "导出"],
            DocumentIntent.MERGE: ["合并", "整合", "拼接"],
            DocumentIntent.ANALYZE: ["分析", "提取", "统计"]
        }
        
        # 匹配意图
        for intent, keywords in intent_keywords.items():
            if any(kw in command for kw in keywords):
                return intent
        
        return DocumentIntent.GENERATE  # 默认生成
    
    def _execute_intent(self, intent: DocumentIntent, params: Dict) -> Dict:
        """执行文档处理"""
        handlers = {
            DocumentIntent.GENERATE: self._handle_generate,
            DocumentIntent.READ: self._handle_read,
            DocumentIntent.MODIFY: self._handle_modify,
            DocumentIntent.CONVERT: self._handle_convert,
            DocumentIntent.MERGE: self._handle_merge,
            DocumentIntent.ANALYZE: self._handle_analyze
        }
        
        return handlers[intent](params)
```
