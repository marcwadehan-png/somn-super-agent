## 调度器信息

| 属性 | 值 |
|-----|---|
| **调度器ID** | SD-F2 |
| **内部名称** | FourLevelDispatchController |
| **类型** | 总控调度器 (L0) |

## 核心调度代码

```python
import time
import uuid
from typing import Dict, List, Optional, Tuple, Any
from dataclasses import dataclass, field
from enum import Enum
from abc import ABC, abstractmethod


class Level(Enum):
    """调度级别枚举"""
    L1_INSTINCT = "L1"
    L2_LOGIC = "L2"
    L3_WISDOM = "L3"
    L4_INTUITION = "L4"


class DispatcherState(Enum):
    """调度器状态"""
    READY = "ready"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"
    ESCALATING = "escalating"


@dataclass
class DispatchRequest:
    """调度请求"""
    request_id: str = field(default_factory=lambda: str(uuid.uuid4()))
    problem: Dict[str, Any] = field(default_factory=dict)
    level: Optional[Level] = None
    forced_level: Optional[Level] = None
    allow_escalation: bool = True
    max_time: float = 30.0
    metadata: Dict[str, Any] = field(default_factory=dict)
    trace_enabled: bool = True


@dataclass
class DispatchResponse:
    """调度响应"""
    request_id: str
    dispatcher_id: str = "SD-F2"
    level: Level
    time_spent: float
    dispatched_to: List[str]
    confidence: float
    output: Dict[str, Any]
    state: DispatcherState = DispatcherState.COMPLETED
    escalated: bool = False
    escalation_target: Optional[Level] = None
    trace: Dict[str, Any] = field(default_factory=dict)
    error: Optional[str] = None


class BaseDispatcher(ABC):
    """调度器基类"""
    
    dispatcher_id: str = "BASE"
    
    def __init__(self):
        self.state = DispatcherState.READY
        self._metrics = {
            "total_calls": 0,
            "failed_calls": 0,
            "avg_time": 0.0
        }
    
    @abstractmethod
    def _execute(self, request: DispatchRequest) -> Dict[str, Any]:
        """执行调度逻辑"""
        pass
    
    def dispatch(self, request: DispatchRequest) -> DispatchResponse:
        """统一调度入口"""
        self.state = DispatcherState.RUNNING
        self._metrics["total_calls"] += 1
        start_time = time.time()
        
        try:
            output = self._execute(request)
            self.state = DispatcherState.COMPLETED
            return self._build_response(request, output, time.time() - start_time)
        except Exception as e:
            self.state = DispatcherState.FAILED
            self._metrics["failed_calls"] += 1
            return self._build_error_response(request, str(e))
    
    def _build_response(self, request: DispatchRequest, 
                       output: Dict, elapsed: float) -> DispatchResponse:
        """构建响应"""
        return DispatchResponse(
            request_id=request.request_id,
            dispatcher_id=self.dispatcher_id,
            level=getattr(request, 'level', Level.L2_LOGIC),
            time_spent=elapsed,
            dispatched_to=output.get("dispatched_to", [self.dispatcher_id]),
            confidence=output.get("confidence", 0.8),
            output=output,
            state=self.state,
            trace=output.get("trace", {})
        )
    
    def _build_error_response(self, request: DispatchRequest, 
                             error: str) -> DispatchResponse:
        """构建错误响应"""
        return DispatchResponse(
            request_id=request.request_id,
            dispatcher_id=self.dispatcher_id,
            level=getattr(request, 'level', Level.L2_LOGIC),
            time_spent=0.0,
            dispatched_to=[],
            confidence=0.0,
            output={},
            state=DispatcherState.FAILED,
            error=error
        )
    
    def get_state(self) -> DispatcherState:
        return self.state
    
    def get_metrics(self) -> Dict:
        return self._metrics.copy()


class LevelAssessor:
    """级别评估器"""
    
    # 关键词权重配置
    KEYWORD_WEIGHTS = {
        # L1关键词
        "L1": ["是什么", "如何", "怎么", "告诉我", "查询", "时间", "天气", 
               "紧急", "马上", "立刻", "立即", "现在"],
        # L2关键词
        "L2": ["分析", "推理", "判断", "论证", "评估", "比较", "数据", 
               "统计", "检验", "验证", "逻辑"],
        # L3关键词
        "L3": ["战略", "规划", "转型", "策划", "方案", "系统", "全面", 
               "深度", "复杂", "综合", "融合", "整合"],
        # L4关键词
        "L4": ["创新", "突破", "颠覆", "创造", "根本", "本质", "核心", 
               "理论", "哲学", "智慧", "直觉", "极致"]
    }
    
    # 复杂度评分
    COMPLEXITY_SCORES = {
        "question_length": {"short": 1, "medium": 2, "long": 3},
        "domain_count": {1: 1, 2: 2, 3: 3},
        "time_horizon": {"day": 1, "month": 2, "year": 3, "decade": 4}
    }
    
    @classmethod
    def assess(cls, problem: Dict[str, Any]) -> Tuple[Level, float]:
        """
        评估问题级别
        
        Returns:
            (level, confidence) 元组
        """
        description = problem.get("description", "").lower()
        context = problem.get("context", "")
        keywords = problem.get("keywords", [])
        
        # 统计各级别关键词命中
        scores = {"L1": 0, "L2": 0, "L3": 0, "L4": 0}
        
        for level, kws in cls.KEYWORD_WEIGHTS.items():
            for kw in kws:
                if kw in description or kw in context:
                    scores[level] += 1
        
        # 计算复杂度加成
        complexity_bonus = cls._calculate_complexity(problem)
        
        # 综合评分
        total = sum(scores.values()) + complexity_bonus
        if total == 0:
            return Level.L2_LOGIC, 0.7  # 默认L2
        
        # 归一化并决定级别
        max_level = max(scores, key=scores.get)
        max_score = scores[max_level]
        
        # 计算置信度
        confidence = min(0.95, max_score / (total + 1) + 0.5)
        
        # 级别映射
        level_map = {
            "L1": Level.L1_INSTINCT,
            "L2": Level.L2_LOGIC,
            "L3": Level.L3_WISDOM,
            "L4": Level.L4_INTUITION
        }
        
        return level_map.get(max_level, Level.L2_LOGIC), confidence
    
    @classmethod
    def _calculate_complexity(cls, problem: Dict) -> int:
        """计算复杂度加成"""
        bonus = 0
        
        # 问题长度
        desc_len = len(problem.get("description", ""))
        if desc_len < 50:
            bonus += 1
        elif desc_len > 200:
            bonus += 3
        
        # 多关键词
        if len(problem.get("keywords", [])) > 3:
            bonus += 2
        
        return bonus


class FourLevelDispatchController(BaseDispatcher):
    """
    SD-F2 四级调度总控引擎
    
    这是SageDispatch的核心入口调度器，负责：
    1. 智能评估问题复杂度
    2. 选择最优处理级别
    3. 协调各专业调度器
    4. 处理级别升级
    """
    
    dispatcher_id: str = "SD-F2"
    
    def __init__(self, config: Dict = None):
        super().__init__()
        self.config = config or {}
        self.assessor = LevelAssessor()
        
        # 调度器缓存
        self._dispatchers: Dict[str, BaseDispatcher] = {}
        
        # 级别配置
        self.LEVEL_CONFIG = {
            Level.L1_INSTINCT: {
                "name": "本能直觉",
                "time_limit": 0.1,
                "target_cells": ["SD-P1"],
                "escalate_if": lambda r, o: o.get("confidence", 1.0) < 0.5
            },
            Level.L2_LOGIC: {
                "name": "逻辑推理",
                "time_limit": 1.0,
                "target_cells": ["SD-P1", "SD-R2"],
                "escalate_if": lambda r, o: (
                    o.get("confidence", 1.0) < 0.6 or 
                    r.problem.get("requires_wisdom", False)
                )
            },
            Level.L3_WISDOM: {
                "name": "智慧融合",
                "time_limit": 10.0,
                "target_cells": ["SD-F1", "SD-P1"],
                "escalate_if": lambda r, o: (
                    o.get("confidence", 1.0) < 0.5 or
                    r.problem.get("is_innovative", False)
                )
            },
            Level.L4_INTUITION: {
                "name": "极致直觉",
                "time_limit": 30.0,
                "target_cells": ["SD-F1", "SD-D3", "SD-C2"],
                "escalate_if": lambda r, o: False
            }
        }
    
    def _execute(self, request: DispatchRequest) -> Dict[str, Any]:
        """执行四级调度"""
        trace = {"states": [], "sub_dispatches": []}
        
        # 1. 确定调度级别
        if request.forced_level:
            level = request.forced_level
            trace["level_decision"] = "forced"
        else:
            level, confidence = self.assessor.assess(request.problem)
            trace["level_decision"] = {
                "assessed_level": level.value,
                "confidence": confidence
            }
        
        trace["states"].append(f"LEVEL_SELECTED:{level.value}")
        
        # 2. 执行该级别调度
        outputs = []
        target_cells = self.LEVEL_CONFIG[level]["target_cells"]
        
        for cell_id in target_cells:
            trace["states"].append(f"DISPATCHING:{cell_id}")
            result = self._dispatch_to_cell(cell_id, request, level)
            outputs.append(result)
            trace["sub_dispatches"].append({
                "cell_id": cell_id,
                "result": result
            })
        
        # 3. 综合结果
        combined_output = self._combine_outputs(outputs)
        
        # 4. 检查是否需要升级
        config = self.LEVEL_CONFIG[level]
        if request.allow_escalation and config["escalate_if"](request, combined_output):
            trace["states"].append("ESCALATING")
            escalated = self._execute_escalation(request, level)
            combined_output.update(escalated)
        
        trace["states"].append("COMPLETED")
        
        return {
            "level": level.value,
            "dispatched_to": target_cells,
            "confidence": combined_output.get("confidence", 0.8),
            "output": combined_output,
            "trace": trace
        }
    
    def _dispatch_to_cell(self, cell_id: str, request: DispatchRequest, 
                         level: Level) -> Dict[str, Any]:
        """调度到指定格子"""
        # 根据cell_id获取对应调度器
        dispatcher = self._get_dispatcher(cell_id)
        
        # 创建子请求
        sub_request = DispatchRequest(
            request_id=f"{request.request_id}-{cell_id}",
            problem=request.problem,
            level=level,
            max_time=request.max_time,
            metadata={**request.metadata, "parent_cell": "SD-F2"}
        )
        
        return dispatcher.dispatch(sub_request).__dict__
    
    def _get_dispatcher(self, cell_id: str) -> BaseDispatcher:
        """获取调度器实例"""
        if cell_id not in self._dispatchers:
            self._dispatchers[cell_id] = self._create_dispatcher(cell_id)
        return self._dispatchers[cell_id]
    
    def _create_dispatcher(self, cell_id: str) -> BaseDispatcher:
        """创建调度器实例"""
        dispatcher_map = {
            "SD-P1": lambda: ProblemDispatcher(),
            "SD-R2": lambda: FallacyChecker(),
            "SD-F1": lambda: SchoolFusion(),
            "SD-D1": lambda: SuperReasoning(mode="light"),
            "SD-D2": lambda: SuperReasoning(mode="standard"),
            "SD-D3": lambda: SuperReasoning(mode="deep"),
            "SD-C1": lambda: YinYangDecision(),
            "SD-C2": lambda: DivineArchitecture(),
            "SD-E1": lambda: ChainExecutor(),
        }
        
        factory = dispatcher_map.get(cell_id)
        if factory:
            return factory()
        
        # 默认返回空调度器
        return BaseDispatcher()
    
    def _combine_outputs(self, outputs: List[Dict]) -> Dict[str, Any]:
        """合并多个调度结果"""
        if not outputs:
            return {"confidence": 0.5, "output": {}}
        
        # 提取所有输出
        all_outputs = []
        confidences = []
        
        for out in outputs:
            if isinstance(out, dict):
                all_outputs.append(out.get("output", {}))
                confidences.append(out.get("confidence", 0.8))
        
        # 加权平均置信度
        avg_confidence = sum(confidences) / len(confidences) if confidences else 0.8
        
        return {
            "confidence": avg_confidence,
            "output": all_outputs[0] if len(all_outputs) == 1 else all_outputs,
            "count": len(outputs)
        }
    
    def _execute_escalation(self, request: DispatchRequest, 
                           current_level: Level) -> Dict[str, Any]:
        """执行级别升级"""
        # 定义升级路径
        escalation_map = {
            Level.L1_INSTINCT: Level.L2_LOGIC,
            Level.L2_LOGIC: Level.L3_WISDOM,
            Level.L3_WISDOM: Level.L4_INTUITION,
            Level.L4_INTUITION: Level.L4_INTUITION  # 最高级不升级
        }
        
        next_level = escalation_map.get(current_level, current_level)
        
        if next_level == current_level:
            return {"escalated": False}
        
        # 执行升级调度
        target_cells = self.LEVEL_CONFIG[next_level]["target_cells"]
        escalated_results = []
        
        for cell_id in target_cells:
            result = self._dispatch_to_cell(cell_id, request, next_level)
            escalated_results.append(result)
        
        return {
            "escalated": True,
            "original_level": current_level.value,
            "escalated_to": next_level.value,
            "escalated_results": escalated_results
        }

## 状态机

```
READY 
   │
   ▼ (dispatch调用)
RUNNING ─────────────────────────────────────────┐
   │                                             │
