# A1.2 谬误检测调度引擎

## 调度器信息
- **类型**: 逻辑校验调度引擎
- **调度层级**: L2（自动嵌入）
- **触发条件**: A1.1推理时自动触发 / 主动调用
- **性能目标**: <10ms（8项检测）
- **依赖格子**: 无（独立模块）
- **优先级**: P1

---

## 核心调度代码

```python
import re
from typing import List, Dict, Tuple, Optional
from dataclasses import dataclass
from enum import Enum


class FallacyType(Enum):
    """谬误类型枚举"""
    AD_HOMINEM = "ad_hominem"
    STRAWMAN = "strawman"
    SLIPPERY_SLOPE = "slippery_slope"
    CIRCULAR_REASONING = "circular_reasoning"
    APPEAL_TO_AUTHORITY = "appeal_to_authority"
    APPEAL_TO_EMOTION = "appeal_to_emotion"
    FALSE_DILEMMA = "false_dilemma"
    HASTY_GENERALIZATION = "hasty_generalization"
    NONE = "none"


@dataclass
class FallacyMatch:
    """谬误匹配结果"""
    fallacy_type: FallacyType
    position_start: int
    position_end: int
    matched_text: str
    confidence: float
    suggestion: str

    def to_dict(self) -> Dict:
        return {
            "type": self.fallacy_type.value,
            "position": self.position_start,
            "text": self.matched_text,
            "confidence": self.confidence,
            "suggestion": self.suggestion
        }


@dataclass
class FallacyReport:
    """谬误检测报告"""
    matches: List[FallacyMatch]
    overall_score: float
    passed: bool
    argument_quality: str
    penalties: Dict[str, float]


class FallacyChecker:
    """
    A1.2 谬误检测调度引擎

    调度参数:
        threshold: 触发阈值 (0.0-1.0)
        strict_mode: 严格模式（降低阈值）
        auto_fix: 是否生成修正建议
        include_soft: 是否包含软检测（低置信度）
    """

    def __init__(self, config: Dict = None):
        self.config = config or {}
        self.threshold = self.config.get("threshold", 0.7)
        self.strict_mode = self.config.get("strict_mode", False)
        self.auto_fix = self.config.get("auto_fix", False)

        if self.strict_mode:
            self.threshold *= 0.8

        # 编译正则表达式
        self._compile_patterns()

    def _compile_patterns(self):
        """编译所有检测模式"""
        self.detectors = {}
        for fallacy_id, config in FallacyDetector.FALLACY_REGISTRY.items():
            self.detectors[fallacy_id] = {
                "name": config["name"],
                "patterns": [re.compile(p, re.IGNORECASE) for p in config["patterns"]],
                "severity": config["severity"],
                "fix": config["fix"]
            }

    def check(self, argument: str | List[str]) -> List[FallacyMatch]:
        """
        主检测入口

        Args:
            argument: 待检测论证（字符串或句子列表）

        Returns:
            List[FallacyMatch]: 检测到的谬误列表
        """
        matches = []

        # 转换为字符串列表
        if isinstance(argument, str):
            sentences = self._split_sentences(argument)
        else:
            sentences = argument

        # 遍历每个检测器
        for fallacy_id, detector in self.detectors.items():
            for sentence in sentences:
                match = self._check_single(sentence, fallacy_id, detector)
                if match:
                    matches.append(match)

        # 按置信度排序
        matches.sort(key=lambda m: m.confidence, reverse=True)

        return matches

    def _check_single(self, text: str, fallacy_id: str, detector: Dict) -> Optional[FallacyMatch]:
        """单条检测"""
        for pattern in detector["patterns"]:
            match = pattern.search(text)
            if match:
                return FallacyMatch(
                    fallacy_type=FallacyType(fallacy_id),
                    position_start=match.start(),
                    position_end=match.end(),
                    matched_text=match.group(),
                    confidence=detector["severity"] * 0.9,  # 略微降低以留有余地
                    suggestion=detector["fix"]
                )
        return None

    def _split_sentences(self, text: str) -> List[str]:
        """分句"""
        # 按常见分隔符分句
        sentences = re.split(r'[。！？；\n]', text)
        return [s.strip() for s in sentences if s.strip()]

    def generate_report(self, argument: str | List[str]) -> FallacyReport:
        """
        生成完整检测报告

        Args:
            argument: 待检测论证

        Returns:
            FallacyReport: 完整报告
        """
        matches = self.check(argument)

        # 计算总分
        overall_score = 1.0
        penalties = {}

        for match in matches:
            if match.confidence >= self.threshold:
                penalty = match.confidence * 0.1
                penalties[match.fallacy_type.value] = penalty
                overall_score -= penalty

        overall_score = max(0, overall_score)

        # 判断是否通过
        passed = overall_score >= 0.7

        # 评估论证质量
        if overall_score >= 0.9:
            quality = "优秀：无明显逻辑谬误"
        elif overall_score >= 0.7:
            quality = "良好：存在少量可改进空间"
        elif overall_score >= 0.5:
            quality = "一般：存在明显逻辑问题"
        else:
            quality = "较差：存在严重逻辑谬误"

        return FallacyReport(
            matches=matches,
            overall_score=overall_score,
            passed=passed,
            argument_quality=quality,
            penalties=penalties
        )

    def check_with_fix(self, argument: str | List[str]) -> Tuple[FallacyReport, List[str]]:
        """
        检测并生成修正建议

        Returns:
            Tuple[FallacyReport, List[str]]: 报告和修正后的论证
        """
        report = self.generate_report(argument)
        fixed_arguments = []

        if isinstance(argument, str):
            argument = self._split_sentences(argument)

        for match in report.matches:
            if match.confidence >= self.threshold:
                # 生成修正后的句子
                fixed = self._apply_fix(argument, match)
                if fixed:
                    fixed_arguments.append(fixed)

        return report, fixed_arguments

    def _apply_fix(self, sentences: List[str], match: FallacyMatch) -> Optional[str]:
        """应用修正"""
        if match.position_start >= len(sentences):
            return None

        sentence = sentences[match.position_start]
        fix_template = f"[原句: {match.matched_text}] → {match.suggestion}"

        return fix_template
```

## 调度参数详解

| 参数 | 类型 | 默认值 | 说明 |
|-----|------|-------|------|
| `threshold` | float | 0.7 | 触发阈值 |
| `strict_mode` | bool | False | 严格模式(阈值*0.8) |
| `auto_fix` | bool | False | 生成修正建议 |
| `include_soft` | bool | True | 包含软检测 |
