```python
    @classmethod
    def get_by_keywords(cls, keywords: List[str]) -> List[Tuple[str, float]]:
        """根据关键词匹配学派"""
        matches = []
        for school_id, school in cls.SCHOOLS.items():
            score = 0
            school_keywords = school.get("keywords", [])
            for kw in keywords:
                if kw in school_keywords:
                    score += 1
            if score > 0:
                matches.append((school_id, score / len(school_keywords)))
        return sorted(matches, key=lambda x: x[1], reverse=True)

    @classmethod
    def get_by_scenario(cls, scenario: str) -> List[Tuple[str, float]]:
        """根据场景匹配学派"""
        matches = []
        for school_id, school in cls.SCHOOLS.items():
            scenarios = school.get("scenario", [])
            if scenario in scenarios:
                matches.append((school_id, school.get("weight_default", 0.5)))
        return sorted(matches, key=lambda x: x[1], reverse=True)
```

## 核心调度代码

```python
from typing import List, Dict, Tuple, Optional, Set
from dataclasses import dataclass, field
from enum import Enum


class FusionMode(Enum):
    """融合模式"""
    PARALLEL = "parallel"      # 并行融合
    SEQUENTIAL = "sequential"  # 顺序融合
    HIERARCHICAL = "hierarchical"  # 层级融合


@dataclass
class SchoolContribution:
    """学派贡献"""
    school_id: str
    school_name: str
    contribution: float  # 贡献度 0-1
    perspective: str     # 提供的视角
    key_insight: str     # 关键洞见
    recommendations: List[str] = field(default_factory=list)


@dataclass
class FusionConflict:
    """融合冲突"""
    school_a: str
    school_b: str
    conflict_type: str
    resolution: str
    recommendation: str


@dataclass
class FusionResult:
    """融合结果"""
    primary_schools: List[SchoolContribution]
    secondary_schools: List[SchoolContribution]
    synthesis: str
    conflicts: List[FusionConflict]
    action_recommendations: List[str]
    confidence: float


class SchoolFusionEngine:
    """
    A2.1 25学派融合调度引擎

    调度参数:
        school_count: 融合学派数量 (默认3-5)
        complexity: 问题复杂度 (1-5)
        focus_area: 重点领域
        fusion_mode: 融合模式
    """

    def __init__(self, config: Dict = None):
        self.config = config or {}
        self.school_count = self.config.get("school_count", 5)
        self.complexity = self.config.get("complexity", 3)
        self.focus_area = self.config.get("focus_area", "strategy")
        self.fusion_mode = self.config.get("fusion_mode", FusionMode.PARALLEL)

    def dispatch(self, problem: Dict) -> FusionResult:
        """
        主调度入口

        Args:
            problem: {
                "description": "问题描述",
                "keywords": ["关键词1", "关键词2"],
                "context": "背景上下文",
                "scenario": "适用场景"
            }

        Returns:
            FusionResult: 融合结果
        """
        # 1. 分析问题类型
        problem_analysis = self._analyze_problem(problem)

        # 2. 选择学派集群
        school_clusters = self._select_school_clusters(problem_analysis)

        # 3. 获取学派视角
        perspectives = self._get_perspectives(school_clusters, problem)

        # 4. 识别和处理冲突
        conflicts = self._resolve_conflicts(school_clusters)

        # 5. 综合输出
        synthesis = self._generate_synthesis(perspectives, conflicts, problem)

        # 6. 生成行动建议
        actions = self._generate_actions(perspectives, synthesis)

        return FusionResult(
            primary_schools=perspectives["primary"],
            secondary_schools=perspectives["secondary"],
            synthesis=synthesis,
            conflicts=conflicts,
            action_recommendations=actions,
            confidence=self._calculate_confidence(perspectives)
        )

    def _analyze_problem(self, problem: Dict) -> Dict:
        """分析问题"""
        keywords = problem.get("keywords", [])
        description = problem.get("description", "")

        # 提取问题类型
        problem_types = []
        if any(k in description for k in ["战略", "方向", "规划"]):
            problem_types.append("strategic")
        if any(k in description for k in ["竞争", "博弈", "谈判"]):
            problem_types.append("competitive")
        if any(k in description for k in ["创新", "突破", "变革"]):
            problem_types.append("innovative")
        if any(k in description for k in ["执行", "落地", "实施"]):
            problem_types.append("execution")

        return {
            "types": problem_types,
            "keywords": keywords,
            "scenario": problem.get("scenario", "general")
        }

    def _select_school_clusters(self, analysis: Dict) -> Dict:
        """选择学派集群"""
        primary = []
        secondary = []
        contrast = []

        # 根据关键词匹配
        by_keywords = SchoolRegistry.get_by_keywords(analysis["keywords"])

        # 根据场景匹配
        by_scenario = SchoolRegistry.get_by_scenario(analysis.get("scenario", ""))

        # 综合评分
        school_scores = {}
        for school_id, score in by_keywords:
            school_scores[school_id] = school_scores.get(school_id, 0) + score * 0.6
        for school_id, score in by_scenario:
            school_scores[school_id] = school_scores.get(school_id, 0) + score * 0.4

        # 排序并分组
        sorted_schools = sorted(school_scores.items(), key=lambda x: x[1], reverse=True)

        primary_count = min(3, len(sorted_schools))
        for i, (school_id, _) in enumerate(sorted_schools[:primary_count]):
            primary.append(school_id)

        for school_id, _ in sorted_schools[primary_count:primary_count+3]:
            secondary.append(school_id)

        # 对比学派（用于反驳）
        for school_id, _ in sorted_schools[-2:]:
            contrast.append(school_id)

        return {
            "primary": primary,
            "secondary": secondary,
            "contrast": contrast,
            "all_selected": primary + secondary + contrast
        }

    def _get_perspectives(self, clusters: Dict, problem: Dict) -> Dict:
        """获取各学派视角"""
        primary_perspectives = []
        secondary_perspectives = []

        for school_id in clusters["primary"]:
            school = SchoolRegistry.SCHOOLS.get(school_id)
            if school:
                perspective = self._generate_school_perspective(school, problem)
                primary_perspectives.append(perspective)

        for school_id in clusters["secondary"]:
            school = SchoolRegistry.SCHOOLS.get(school_id)
            if school:
                perspective = self._generate_school_perspective(school, problem)
                secondary_perspectives.append(perspective)

        return {
            "primary": primary_perspectives,
            "secondary": secondary_perspectives
        }

    def _generate_school_perspective(self, school: Dict, problem: Dict) -> SchoolContribution:
        """生成单个学派视角"""
        description = problem.get("description", "")

        return SchoolContribution(
            school_id=school.get("name", ""),
            school_name=school["name"],
            contribution=school.get("weight_default", 0.5),
            perspective=f"从{school['name']}视角分析：{school.get('methods', [])}",
            key_insight=f"{school['name']}认为：{school.get('questions', ['如何解决问题？'])[0]}",
            recommendations=school.get("methods", [])[:2]
        )

    def _resolve_conflicts(self, clusters: Dict) -> List[FusionConflict]:
        """解决学派冲突"""
        conflicts = []

        # 定义已知冲突
        conflict_map = {
            ("DAO", "MGMT"): ("自然无为 vs 目标管理", "在长期目标和短期执行间平衡"),
            ("FA", "RU"): ("严刑峻法 vs 以德服人", "法治与德治结合"),
            ("SCI", "FO"): ("实证验证 vs 缘起性空", "在现实约束下追求真相"),
        }

        for school_a in clusters["primary"]:
            for school_b in clusters["primary"]:
                if school_a < school_b:
                    key = (school_a, school_b)
                    reverse_key = (school_b, school_a)
                    if key in conflict_map:
                        conflicts.append(FusionConflict(
                            school_a=school_a,
                            school_b=school_b,
                            conflict_type=conflict_map[key][0],
                            resolution=conflict_map[key][1],
                            recommendation=""
                        ))

        return conflicts

    def _generate_synthesis(self, perspectives: Dict, conflicts: List[FusionConflict],
                           problem: Dict) -> str:
        """生成综合视角"""
        lines = [f"【{problem.get('description', '问题')}的综合分析】"]

        lines.append("\n核心视角：")
        for p in perspectives["primary"][:3]:
            lines.append(f"- {p.school_name}：{p.key_insight}")

        if conflicts:
            lines.append("\n需要平衡的张力：")
            for c in conflicts:
                lines.append(f"- {c.school_a} vs {c.school_b}：{c.conflict_type}")
                lines.append(f"  → {c.resolution}")

        lines.append("\n综合建议：")
        lines.append("在多元视角整合下，建议采取平衡策略...")

        return "\n".join(lines)

    def _generate_actions(self, perspectives: Dict, synthesis: str) -> List[str]:
        """生成行动建议"""
        actions = []
        for p in perspectives["primary"]:
            actions.extend(p.recommendations)
        return list(set(actions))[:5]

    def _calculate_confidence(self, perspectives: Dict) -> float:
        """计算置信度"""
        if not perspectives["primary"]:
            return 0.5

        avg_contribution = sum(p.contribution for p in perspectives["primary"]) / len(perspectives["primary"])
        return min(avg_contribution * 1.2, 1.0)
```
