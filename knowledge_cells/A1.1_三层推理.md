# A1.1 三层推理调度引擎

## 调度器信息
- **类型**: 推理调度引擎
- **调度层级**: L2（自动调用）
- **触发条件**: 需要结构化推理的问题
- **性能目标**: <100ms（基础）、<500ms（深度）
- **依赖格子**: A1.2, A7
- **优先级**: P1

---

## 核心调度架构

```
                    ┌─────────────────────────────────┐
                    │     A1.1 三层推理调度引擎        │
                    │     Three-Layer Reasoning Engine │
                    └────────────────┬────────────────┘
                                     │
          ┌──────────────────────────┼──────────────────────────┐
          │                          │                          │
          ▼                          ▼                          ▼
   ┌─────────────┐           ┌─────────────┐           ┌─────────────┐
   │  Layer-1    │           │  Layer-2    │           │  Layer-3    │
   │  前提提取   │──────────▶│  推理执行   │──────────▶│  结论评估   │
```

## 核心调度代码

```python
class ThreeLayerReasoningEngine:
    """
    A1.1 三层推理调度引擎

    调度参数:
        mode: 推理模式 (deductive/inductive/analogical/counterfactual)
        depth: 推理深度 (1-3)
        fallacy_check: 是否启用谬误检测
        auto_escalate: 是否自动升级
    """

    # 状态枚举
    class State(Enum):
        INIT = "init"
        PREMISE_EXTRACTING = "premise_extracting"
        PREMISE_ERROR = "premise_error"
        INFERENCING = "inferencing"
        EVALUATING = "evaluating"
        COMPLETED = "completed"
        FAILED = "failed"

    # 推理模式配置
    REASONING_MODES = {
        "deductive": {
            "description": "演绎推理（必然）",
            "certainty": 1.0,
            "requires": ["全称前提", "肯定前件"],
            "examples": ["所有A是B", "C是A", "∴ C是B"]
        },
        "inductive": {
            "description": "归纳推理（概然）",
            "certainty": 0.7,
            "requires": ["样本充分", "无反例"],
            "examples": ["观察到A1,A2,A3都是B", "∴ 所有A可能是B"]
        },
        "analogical": {
            "description": "类比推理（迁移）",
            "certainty": 0.5,
            "requires": ["相似性高", "相关属性"],
            "examples": ["A有属性X,Y", "B也有X,Y", "∴ B可能有属性Z"]
        },
        "counterfactual": {
            "description": "反事实推理（假设）",
            "certainty": 0.6,
            "requires": ["假设前提", "因果链"],
            "examples": ["如果没有X", "则可能不会发生Y"]
        }
    }

    def __init__(self, config: Dict = None):
        self.config = config or {}
        self.state = self.State.INIT
        self.layer_outputs = {}
        self.mode = self.config.get("mode", "deductive")
        self.depth = self.config.get("depth", 1)
        self.fallacy_check = self.config.get("fallacy_check", True)
        self.auto_escalate = self.config.get("auto_escalate", True)

        # 调度追踪
        self.trace = {
            "state_history": [],
            "layer_times": {},
            "fallacies_detected": []
        }

    def dispatch(self, input_data: Dict) -> "DispatchResult":
        """
        主调度入口

        Args:
            input_data: {
                "premises": [...],      # 前提列表
                "context": "...",        # 上下文
                "question": "...",       # 问题
                "expected_mode": "..."   # 期望的推理模式
            }

        Returns:
            DispatchResult: 包含三层推理结果
        """
        start_time = time.time()
        self._transition_to(self.State.PREMISE_EXTRACTING)

        try:
            # Layer 1: 前提提取
            premises = self._layer1_premise_extraction(input_data)
            self.layer_outputs["premises"] = premises

            # Layer 2: 推理执行
            self._transition_to(self.State.INFERENCING)
            reasoning = self._layer2_inference(premises, input_data)
            self.layer_outputs["reasoning"] = reasoning

            # Layer 3: 结论评估
            self._transition_to(self.State.EVALUATING)
            evaluation = self._layer3_evaluation(reasoning, premises)
            self.layer_outputs["evaluation"] = evaluation

            # 完成
            self._transition_to(self.State.COMPLETED)

            return self._build_result(
                success=True,
                premises=premises,
                reasoning=reasoning,
                evaluation=evaluation,
                elapsed=time.time() - start_time
            )

        except ReasoningError as e:
            self._transition_to(self.State.FAILED)
            return self._build_result(success=False, error=str(e))

    def _layer1_premise_extraction(self, input_data: Dict) -> Dict:
        """Layer 1: 前提提取"""
        layer_start = time.time()

        premises = {
            "facts": [],
            "assumptions": [],
            "unknowns": [],
            "sources": [],
            "validity_scores": {}
        }

        # 1.1 事实提取
        raw_premises = input_data.get("premises", [])
        for p in raw_premises:
            if self._is_fact(p):
                premises["facts"].append(p)
                premises["validity_scores"][p["text"]] = self._validate_fact(p)
            else:
                premises["assumptions"].append(p)

        # 1.2 假设识别
        premises["assumptions"].extend(self._extract_assumptions(input_data))

        # 1.3 来源验证
        for fact in premises["facts"]:
            source = fact.get("source", "unknown")
            premises["sources"].append({
                "source": source,
                "verified": self._validate_source(source)
            })

        # 1.4 未知识别
        premises["unknowns"] = self._identify_unknowns(input_data)

        self.trace["layer_times"]["layer1"] = time.time() - layer_start

        return premises

    def _layer2_inference(self, premises: Dict, input_data: Dict) -> Dict:
        """Layer 2: 推理执行"""
        layer_start = time.time()

        reasoning = {
            "mode": self.mode,
            "mode_config": self.REASONING_MODES[self.mode],
            "steps": [],
            "chain": [],
            "intermediate_conclusions": []
        }

        # 2.1 选择推理模式
        if input_data.get("expected_mode"):
            self.mode = input_data["expected_mode"]

        # 2.2 构建推理链
        if self.mode == "deductive":
            reasoning = self._deductive_inference(premises, reasoning)
        elif self.mode == "inductive":
            reasoning = self._inductive_inference(premises, reasoning)
        elif self.mode == "analogical":
            reasoning = self._analogical_inference(premises, reasoning)
        elif self.mode == "counterfactual":
            reasoning = self._counterfactual_inference(premises, reasoning)

        # 2.3 谬误检测
        if self.fallacy_check:
            from .A1_2_dispatcher import FallacyChecker
            checker = FallacyChecker()
            fallacies = checker.check(reasoning["chain"])
            reasoning["fallacies_detected"] = fallacies
            self.trace["fallacies_detected"] = fallacies

            if fallacies:
                reasoning["confidence_penalty"] = len(fallacies) * 0.1

        self.trace["layer_times"]["layer2"] = time.time() - layer_start

        return reasoning

    def _layer3_evaluation(self, reasoning: Dict, premises: Dict) -> Dict:
        """Layer 3: 结论评估"""
        layer_start = time.time()

        evaluation = {
            "validity": "unknown",
            "confidence": 0.0,
            "counterexamples": [],
            "limitations": [],
            "qualified_conclusion": None
        }

        # 3.1 必然性评估
        if reasoning["mode"] == "deductive":
            evaluation["validity"] = self._check_deductive_necessity(reasoning, premises)
        else:
            evaluation["validity"] = self._check_probabilistic_validity(reasoning)

        # 3.2 反例搜索
        if self.depth >= 2:
            evaluation["counterexamples"] = self._find_counterexamples(reasoning, premises)

        # 3.3 生成限定结论
        evaluation["qualified_conclusion"] = self._generate_qualified_conclusion(
            reasoning, evaluation
        )

        # 3.4 计算置信度
        evaluation["confidence"] = self._calculate_confidence(reasoning, evaluation)

        self.trace["layer_times"]["layer3"] = time.time() - layer_start

        return evaluation

    # -------------------- 辅助方法 --------------------

    def _is_fact(self, premise: Dict) -> bool:
        """判断是否为事实"""
        return premise.get("type") == "fact" or "evidence" in premise

    def _validate_fact(self, fact: Dict) -> float:
        """验证事实有效性"""
        score = 0.5
        if fact.get("source"):
            score += 0.3
        if fact.get("evidence"):
            score += 0.2
        return min(score, 1.0)

    def _validate_source(self, source: str) -> bool:
        """验证来源可靠性"""
        reliable_sources = ["official", "verified", "peer_reviewed"]
        return any(s in source.lower() for s in reliable_sources)

    def _extract_assumptions(self, input_data: Dict) -> List[Dict]:
        """提取隐含假设"""
        assumptions = []
        return assumptions

    def _identify_unknowns(self, input_data: Dict) -> List[str]:
        """识别未知信息"""
        return []

    def _deductive_inference(self, premises: Dict, reasoning: Dict) -> Dict:
        """演绎推理"""
        reasoning["steps"].append({
            "type": "deductive",
            "description": "从全称命题推导特称命题"
        })
        reasoning["chain"].append("∀x(P(x) → Q(x)), P(a) ⊢ Q(a)")
        return reasoning

    def _inductive_inference(self, premises: Dict, reasoning: Dict) -> Dict:
        """归纳推理"""
        reasoning["steps"].append({
            "type": "inductive",
            "description": "从样本推导一般结论"
        })
        reasoning["chain"].append("P(a₁), P(a₂), ..., P(aₙ) ⊢ ∀xP(x)")
        return reasoning

    def _analogical_inference(self, premises: Dict, reasoning: Dict) -> Dict:
        """类比推理"""
        reasoning["steps"].append({
            "type": "analogical",
            "description": "从相似案例迁移结论"
        })
        return reasoning

    def _counterfactual_inference(self, premises: Dict, reasoning: Dict) -> Dict:
        """反事实推理"""
        reasoning["steps"].append({
            "type": "counterfactual",
            "description": "假设条件下的推理"
        })
        return reasoning

    def _check_deductive_necessity(self, reasoning: Dict, premises: Dict) -> str:
        """检查演绎必然性"""
        if not premises.get("facts"):
            return "前提不充分"
        if reasoning.get("fallacies_detected"):
            return "存在逻辑谬误"
        return "必然成立"

    def _check_probabilistic_validity(self, reasoning: Dict) -> str:
        """检查概然性"""
        return "概然成立（置信度{}）".format(reasoning.get("confidence", 0.7))

    def _find_counterexamples(self, reasoning: Dict, premises: Dict) -> List:
        """搜索反例"""
        return []

    def _generate_qualified_conclusion(self, reasoning: Dict, evaluation: Dict) -> str:
        """生成限定结论"""
        base = reasoning.get("chain", ["无法推导"])[:-1]
        if evaluation.get("counterexamples"):
            return f"{base}（需注意反例）"
        return base

    def _calculate_confidence(self, reasoning: Dict, evaluation: Dict) -> float:
        """计算置信度"""
        base = self.REASONING_MODES.get(self.mode, {}).get("certainty", 0.5)
        penalty = reasoning.get("confidence_penalty", 0)
        return max(0, base - penalty)

    def _transition_to(self, new_state):
        """状态转换"""
        self.trace["state_history"].append({
            "from": self.state.value,
            "to": new_state.value,
            "timestamp": time.time()
        })
        self.state = new_state

    def _build_result(self, success: bool, **kwargs) -> "DispatchResult":
        """构建结果"""
        return DispatchResult(
            dispatcher_id="A1.1",
            level="L2",
            confidence=kwargs.get("evaluation", {}).get("confidence", 0.8),
            output={
                "success": success,
                "state": self.state.value,
                "premises": kwargs.get("premises"),
                "reasoning": kwargs.get("reasoning"),
                "evaluation": kwargs.get("evaluation"),
                "trace": self.trace
            },
            time_elapsed=f"{kwargs.get('elapsed', 0):.3f}s",
            error=kwargs.get("error")
        )


class ReasoningError(Exception):
    """推理异常"""
    pass
```

## 调度参数详解

| 参数 | 类型 | 默认值 | 说明 |
|-----|------|-------|------|
| `mode` | str | "deductive" | 推理模式 |
| `depth` | int | 1 | 推理深度(1-3) |
| `fallacy_check` | bool | True | 启用谬误检测 |
| `auto_escalate` | bool | True | 自动升级到A3.x |
| `timeout` | float | 100 | 超时(ms) |

## 自动升级规则

| 条件 | 升级目标 | 原因 |
|-----|---------|------|
| 推理深度>2 | A3.1 | 需要深度推理 |
| 发现多个谬误 | A3.2 | 需要翰林院审核 |
| 战略/创新问题 | A3.3 | 需要极致深度 |
