## 二、快速开始

### 2.1 方式一：神之架构直接调用（推荐）

```python
# -*- coding: utf-8 -*-
"""
神之架构直接调用 - 智慧调度系统
"""
import sys
import os

# 添加项目路径
sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

from knowledge_cells import dispatch, smart_dispatch, chain, execute_with_track_b

def main():
    # ==================== 基础调度 ====================
    print("=" * 60)
    print("【神之架构】智慧调度系统")
    print("=" * 60)
    
    # 问题示例
    question = "如何提升产品销量？"
    
    # 1. 快捷调度 - 自动选择最优调度器
    print(f"\n📝 问题: {question}")
    print("\n【1】快捷调度 dispatch()")
    result = dispatch(question)
    print(f"结果: {result}")
    
    # 2. 智能调度 - 自动匹配合适的推理级别
    print("\n【2】智能调度 smart_dispatch()")
    result = smart_dispatch(question, level=None)  # 自动选择
    print(f"结果: {result}")
    
    # 3. 链式调度 - 按顺序执行多个调度器
    print("\n【3】链式调度 chain()")
    results = chain(question, ["SD-F1", "SD-D3", "SD-C2"])
    for r in results:
        print(f"  → {r}")
    
    # 4. 神行轨调用 - 执行层调用
    print("\n【4】神行轨调用 execute_with_track_b()")
    result = execute_with_track_b("SD-F1", "融合分析", "礼部")
    print(f"结果: {result}")

if __name__ == "__main__":
    main()
```

## 四、完整工作流示例

```python
# -*- coding: utf-8 -*-
"""
SOMN 完整工作流示例
"""
import sys
import os

sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

from knowledge_cells import (
    dispatch,              # 快捷调度
    smart_dispatch,         # 智能调度
    chain,                  # 链式调度
    EightLayerPipeline,     # 八层管道
    DivineReason,           # 三层推理
    PanWisdomTree,          # 42学派
    execute_with_track_b    # 神行轨
)

class SomnWorkflow:
    """SOMN 完整工作流"""
    
    def __init__(self):
        self.pipeline = EightLayerPipeline(lazy=True)
        self.reasoner = DivineReason()
        self.wisdom_tree = PanWisdomTree()
    
    def solve(self, question: str, mode: str = "auto"):
        """
        解决问题
        
        Args:
            question: 问题描述
            mode: 模式 (auto/quick/deep/expert)
        """
        print("=" * 60)
        print(f"📝 问题: {question}")
        print(f"⚙️ 模式: {mode}")
        print("=" * 60)
        
        # 1. 快捷调度
        print("\n【1】快捷调度...")
        result = dispatch(question)
        print(f"   结果: {result[:100]}...")
        
        # 2. 智能调度
        print("\n【2】智能调度...")
        result = smart_dispatch(question, level=None)
        print(f"   结果: {result[:100]}...")
        
        # 3. 八层管道
        print("\n【3】八层管道处理...")
        pipeline_result = self.pipeline.process(question, level="deep")
        print(f"   输出: {pipeline_result.output[:100]}...")
        print(f"   置信度: {pipeline_result.confidence}")
        
        # 4. 知识库增强
        print("\n【4】DomainNexus 知识库增强...")
        # (由 pipeline 自动调用)
        print(f"   知识增强: {pipeline_result.knowledge_enhanced}")
        
        # 5. 论证闭环
        print("\n【5】RefuteCore 论证验证...")
        # (由 pipeline 自动调用)
        print(f"   论证完成")
        
        # 6. 神行轨执行
        print("\n【6】神行轨执行...")
        exec_result = execute_with_track_b("SD-E1", "工部执行", "工部")
        print(f"   执行: {exec_result}")
        
        return pipeline_result

def main():
    # 创建工作流
    workflow = SomnWorkflow()
    
    # 测试问题
    questions = [
        "分析公司战略转型方案",
        "如何提升产品销量？",
        "人工智能对未来社会的影响"
    ]
    
    for q in questions:
        workflow.solve(q)
        print("\n")

if __name__ == "__main__":
    main()
```
