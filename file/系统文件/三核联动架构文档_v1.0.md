### 6.2 核心调度方法

```python
def three_core_dispatch(self, user_input: str, context: dict) -> DispatchResult:
    """
    三核联动调度主方法
    
    流程: 研究体系(脑) → 神之架构(躯) → 贤者系统(魂) → 执行 → 反馈闭环
    """
    # 第一步：脑·研究体系解析
    research_plan = self.research_framework.parse_and_plan(user_input)
    complexity = research_plan.complexity
    depth = research_plan.depth
    dimensions = research_plan.cross_dimensions
    
    # 第二步：躯·神之架构调度
    main_lines = self.divine_architecture.select_main_lines(complexity)
    positions = self.divine_architecture.allocate_positions(research_plan)
    cross_sync = self.divine_architecture.build_cross_line_sync(main_lines)
    
    # 第三步：魂·贤者系统注入
    wisdom_schools = self.sage_system.match_wisdom_schools(dimensions)
    claws = self.sage_system.activate_claws(positions)
    phase_output = self.sage_system.invoke_phase_chain(depth)
    
    # 第四步：融合执行
    fusion_result = self.fuse_three_core_outputs(
        research_plan, positions, claws, phase_output
    )
    
    # 第五步：反馈闭环
    roi_tracker = self.divine_architecture.get_roi_tracker()
    lifecycle = self.research_framework.get_lifecycle(research_plan.strategy_id)
    
    return DispatchResult(
        result=fusion_result,
        roi_tracker=roi_tracker,
        lifecycle=lifecycle,
        feedback_loop=self.build_feedback_loop(fusion_result)
    )
```
