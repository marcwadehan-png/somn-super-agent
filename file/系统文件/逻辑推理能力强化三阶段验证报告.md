```python
def _select_reasoning_mode(problem: str) -> str:
    if any(kw in problem for kw in ["分析", "推理", "论证", "深入"]):
        return "long_cot_reasoning"
    if any(kw in problem for kw in ["方案", "选择", "比较", "权衡"]):
        return "tot_reasoning"
    if any(kw in problem for kw in ["图", "关系", "网络", "多跳"]):
        return "got_reasoning"
    if any(kw in problem for kw in ["查询", "搜索", "验证", "计算"]):
        return "react_reasoning"
    return "default"
```

### 优化后的选择逻辑（v2.0）

```python
def _select_reasoning_mode(problem) -> str:
    # 优先级1：图关系类 → GoT
    if any(kw in ["图", "关系", "网络", "多跳", "链路", "连接", "关联", "依赖", "拓扑"]):
        return "got_reasoning"
    # 优先级2：方案选择类 → ToT
    if any(kw in ["方案", "选择", "比较", "权衡", "策略", "规划", "路径"]):
        return "tot_reasoning"
    # 优先级3：外部查询类 → ReAct
    if any(kw in ["查询", "搜索", "计算", "查", "获取", "验证", "确认", "检查"]):
        return "react_reasoning"
    # 优先级4：深入分析类 → LongCoT（默认）
    if any(kw in ["分析", "推理", "论证", "深入", "复杂", "多步", "详细", "研究"]):
        return "long_cot_reasoning"
    # 默认
    return "long_cot_reasoning"
```
